/*STR_MTX_GEN 5 sor,10 oszlop,20karakter,2es seed*/
SOROK<-5
OSZLOPOK<-10
SEED<-2
N<-20

STR_MTX_GEN(MTX,SOROK,OSZLOPOK,N,SEED)

MAXDB<-0

for J<-1,OSZLOPOK
  AKTDB <-0
  for I<-1, SOROK
    for K<-1, LENGTH(MTX[I,J])
      if MTX[I,J][K] IN ['A'..'a','a'..'z']
        AKTDB<-AKTDB+1
  if AKTDB>MAXDB
    MAXDB<-AKTDB

Ki: MAXDB
